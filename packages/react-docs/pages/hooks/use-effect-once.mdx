# useEffectOnce

A modified version of the `useEffect` Hook that runs an effect only once.

## Import

```js
import { useEffectOnce } from '@tonic-ui/react-hooks';
```

```js
import { useEffect } from 'react';

const useEffectOnce = (effect) => {
  // eslint-disable-next-line react-hooks/exhaustive-deps
  useEffect(effect, []);
};
```

## Usage

```jsx
function App() {
  const [data, setData] = React.useState(0);

  React.useEffect(() => {
    console.log('useEffect is triggered when data changes', { data });
  }, [data]);

  useEffectOnce(() => {
    console.log('useEffectOnce is triggered only once', { data });
  });

  return (
    <>
      <Box mb="2x">{data}</Box>
      <Button onClick={() => setData(data => data + 1)}>
        Update data
      </Button>
    </>
  );
}
```
